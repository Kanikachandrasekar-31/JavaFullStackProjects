You are building a Notification System. Right now, you can only send a plain text message.
But sometimes you want to add extra features like:
- Sending with Email
- Sending with SMS
- Sending with WhatsApp
------------------------------------------------------------------------------------------------

interface Message{
    String getDescription();//base or component interface
}


class PlainMessage implements Message { //concrete component
    public String getDescription() {
        return "hello";
    }
}


abstract class MessageDecorator implements Message{ //decorator base
    protected Message message;
    public MessageDecorator(Message message) {
        this.message = message;
    }
    public String getDescription() {
        return message.getDescription();
    }
}


class EmailDecorator extends MessageDecorator { //concrete decorators
    public EmailDecorator(Message message) {
        super(message);
    }
    public String getDescription() {
        return super.getDescription() + " + Sending with email";
    }
}

class SMSDecorator extends MessageDecorator { //concrete decorator
    public SMSDecorator(Message message) {
        super(message);
    }
    public String getDescription() {
        return super.getDescription() + " + Sending with SMS";
    }
}
class WhatsappDecorator extends MessageDecorator { //concrete decorator
    public WhatsappDecorator(Message message) {
        super(message);
    }
    public String getDescription() {
        return super.getDescription() + " + Sending with Whatsapp";
    }
}
